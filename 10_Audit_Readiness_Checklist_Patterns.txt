class EvidenceManagementPattern:
    """
    Pattern for managing audit evidence collection and verification.
    Learning Focus: Evidence tracking, version control, and integrity verification.
    """
    def collect_evidence_for_item(self, checklist_item, system_context):
        """
        Pattern: Structured evidence collection for a checklist item.
        """
        evidence_bundle = {
            "checklist_item_id": checklist_item["item_id"],
            "collection_timestamp": "{{TIMESTAMP}}",
            "collected_by": system_context["collector_id"],
            "evidence_sources": []
        }

        for evidence_type in checklist_item["evidence_type"]:
            # Pattern: Type-specific evidence collection
            evidence = self._collect_evidence_by_type(
                evidence_type,
                checklist_item,
                system_context
            )

            if evidence:
                # Pattern: Add integrity verification
                evidence["integrity_hash"] = self._calculate_hash(evidence["content"])
                evidence_bundle["evidence_sources"].append(evidence)

        # Pattern: Overall bundle verification
        evidence_bundle["bundle_hash"] = self._calculate_bundle_hash(evidence_bundle)
        evidence_bundle["collection_status"] = (
            "complete" if len(evidence_bundle["evidence_sources"]) > 0
            else "incomplete"
        )

        return evidence_bundle

    def _collect_evidence_by_type(self, evidence_type, checklist_item, context):
        """Pattern: Abstract evidence collection by type."""
        collectors = {
            "design_documents": self._collect_design_docs,
            "test_reports": self._collect_test_reports,
            "log_files": self._collect_log_files,
            "configuration_files": self._collect_config_files
        }

        collector = collectors.get(evidence_type)
        if not collector:
            return None

        return collector(checklist_item, context)

class ChecklistReviewWorkflowPattern:
    """
    Pattern for managing checklist review and approval processes.
    Learning Focus: Workflow design, role-based approvals, and audit trails.
    """
    def execute_review_workflow(self, checklist, evidence_bundles):
        """
        Pattern: Multi-stage review workflow execution.
        """
        workflow_log = []
        current_state = {
            "checklist": checklist,
            "evidence": evidence_bundles,
            "approvals": {},
            "status": "in_progress"
        }

        # Pattern: Sequential review by role
        for role in checklist["completion_workflow"]["reviewer_chain"]:
            review_result = self._perform_role_review(role, current_state)
            workflow_log.append(review_result)

            current_state["approvals"][role] = review_result["decision"]
            current_state["last_review"] = review_result["timestamp"]

            if review_result["decision"] == "rejected":
                current_state["status"] = "rejected"
                current_state["rejection_reason"] = review_result["comments"]
                break

        if current_state["status"] == "in_progress":
            current_state["status"] = "approved"

        return {
            "final_status": current_state["status"],
            "workflow_log": workflow_log,
            "approval_summary": current_state["approvals"],
            "completion_timestamp": "{{TIMESTAMP}}",
            "note": "Checklist review workflow pattern example."
        }

    def _perform_role_review(self, role, state):
        """Pattern: Role-specific review logic."""
        # In reality, this would interface with a user approval system

class ChecklistVersioningPattern:
    """
    Pattern for versioning checklists and tracking compliance over time.
    Learning Focus: Version control, change tracking, and historical compliance.
    """
    def create_checklist_version(self, base_checklist, changes, rationale):
        """
        Pattern: Create a new version of a checklist with change tracking.
        """
        new_version = {
            "checklist_id": base_checklist["checklist_id"],
            "version": self._increment_version(base_checklist["metadata"]["version"]),
            "based_on_version": base_checklist["metadata"]["version"],
            "created_date": "{{CURRENT_DATE}}",
            "created_by": "{{USER_ID}}",
            "sections": self._apply_changes(base_checklist["sections"], changes),
            "change_log": [
                {
                    "change_id": self._generate_change_id(),
                    "description": change,
                    "rationale": rationale,
                    "timestamp": "{{TIMESTAMP}}"
                }
                for change in changes
            ],
            "metadata": {
                **base_checklist["metadata"],
                "version": self._increment_version(base_checklist["metadata"]["version"]),
                "last_updated": "{{CURRENT_DATE}}"
            }
        }

        # Pattern: Archive previous version
        self._archive_checklist_version(base_checklist)

        return new_version

    def track_compliance_history(self, checklist_results, time_period):
        """
        Pattern: Track compliance scores over time.
        """
        compliance_history = {
            "checklist_id": checklist_results[0]["checklist_id"] if checklist_results else None,
            "time_period": time_period,
            "compliance_metrics": []
        }

        for result in checklist_results:
            completion_rate = self._calculate_completion_rate(result)
            approval_status = result.get("final_status", "unknown")

            compliance_history["compliance_metrics"].append({
                "period": result["completion_timestamp"],
                "completion_rate": completion_rate,
                "approval_status": approval_status,
                "pending_items": self._count_pending_items(result)
            })

        return compliance_history
