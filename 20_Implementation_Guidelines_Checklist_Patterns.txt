class ProcedureDocumentationPattern:
    """
    Pattern: Structured representation of implementation procedures.
    Learning Focus: Procedure decomposition, step organization, and prerequisite management.
    """
    def create_procedure(self, procedure_definition):
        """
        Pattern: Create a structured procedure document from a definition.
        """
        procedure = {
            "procedure_id": f"PROC_{procedure_definition['code']}_{procedure_definition['version']}",
            "title": procedure_definition["title"],
            "description": procedure_definition["description"],
            "version": procedure_definition["version"],
            "last_updated": "{{CURRENT_DATE}}",
            "last_updated_by": "{{USER_ID}}",
            "prerequisites": procedure_definition.get("prerequisites", []),
            "estimated_duration_minutes": procedure_definition.get("estimated_duration", 30),
            "risk_level": procedure_definition.get("risk_level", "medium"),
            "requires_approval": procedure_definition.get("requires_approval", False),
            "approval_role": procedure_definition.get("approval_role"),
            "steps": [],
            "verification_steps": [],
            "rollback_procedure_id": procedure_definition.get("rollback_procedure_id"),
            "related_procedures": procedure_definition.get("related_procedures", [])
        }

        # Pattern: Build procedure steps with sequence
        for i, step_def in enumerate(procedure_definition.get("steps", []), 1):
            step = {
                "step_number": i,
                "step_id": f"{procedure['procedure_id']}_STEP_{i:02d}",
                "title": step_def["title"],
                "description": step_def["description"],
                "command": step_def.get("command"),
                "expected_output": step_def.get("expected_output"),
                "verification_method": step_def.get("verification_method"),
                "timeout_seconds": step_def.get("timeout", 300),
                "critical_step": step_def.get("critical", False),
                "notes": step_def.get("notes", "")
            }
            procedure["steps"].append(step)

        # Pattern: Build verification steps
        for i, verif_def in enumerate(procedure_definition.get("verification_steps", []), 1):
            verification = {
                "verification_id": f"{procedure['procedure_id']}_VERIF_{i:02d}",
                "description": verif_def["description"],
                "method": verif_def["method"],
                "expected_result": verif_def["expected_result"],
                "post_verification": verif_def.get("post_verification", False)
            }
            procedure["verification_steps"].append(verification)

        return procedure

class ProcedureExecutionPattern:
    """
    Pattern: Track execution of implementation procedures.
    Learning Focus: Execution logging, step verification, and audit trail creation.
    """
    def start_procedure_execution(self, procedure_id, context):
        """
        Pattern: Initialize a procedure execution session.
        """
        execution = {
            "execution_id": f"EXEC_{procedure_id}_{{YYYYMMDDHHMMSS}}_{{UUID_SHORT}}",
            "procedure_id": procedure_id,
            "started_at": "{{TIMESTAMP}}",
            "started_by": context.get("actor", "{{USER_ID}}"),
            "environment": context.get("environment", "production"),
            "status": "in_progress",
            "current_step": 0,
            "steps_executed": [],
            "verifications_completed": [],
            "errors": [],
            "notes": context.get("notes", []),
            "completed_at": None,
            "final_status": None
        }

        return execution

    def record_step_execution(self, execution_id, step_number, result, output=None):
        """
        Pattern: Record the execution result of a single procedure step.
        """
        execution = self._get_execution(execution_id)

        step_record = {
            "step_number": step_number,
            "executed_at": "{{TIMESTAMP}}",
            "success": result.get("success", False),
            "output": output or result.get("output"),
            "error": result.get("error"),
            "duration_seconds": result.get("duration_seconds"),
            "verification_status": result.get("verification_status", "pending")
        }

        execution["steps_executed"].append(step_record)
        execution["current_step"] = step_number

        if not step_record["success"]:
            execution["status"] = "failed"
            execution["errors"].append({
                "step": step_number,
                "error": step_record["error"],
                "timestamp": "{{TIMESTAMP}}"
            })

        # Pattern: Check if all steps are complete
        if len(execution["steps_executed"]) == self._get_total_steps(execution["procedure_id"]):
            execution["status"] = "pending_verification"

        return execution

class KnowledgeBasePattern:
    """
    Pattern: Manage FAQ and knowledge base articles for implementation.
    Learning Focus: Content categorization, search, and article lifecycle.
    """
    ARTICLE_CATEGORIES = [
        "getting_started",
        "configuration",
        "troubleshooting",
        "best_practices",
        "reference",
        "faq"
    ]

    def create_article(self, article_definition):
        """
        Pattern: Create a knowledge base article.
        """
        article = {
            "article_id": f"KB_{article_definition['category']}_{{YYYYMMDD}}_{{UUID_SHORT}}",
            "title": article_definition["title"],
            "category": article_definition["category"],
            "tags": article_definition.get("tags", []),
            "author": article_definition.get("author", "{{USER_ID}}"),
            "created_at": "{{TIMESTAMP}}",
            "updated_at": "{{TIMESTAMP}}",
            "version": "1.0",
            "content": {
                "summary": article_definition.get("summary", ""),
                "sections": article_definition.get("sections", [])
            },
            "related_articles": article_definition.get("related_articles", []),
            "related_procedures": article_definition.get("related_procedures", []),
            "helpful_count": 0,
            "not_helpful_count": 0,
            "status": "published"
        }

        return article

    def search_articles(self, query, categories=None, tags=None):
        """
        Pattern: Search knowledge base articles.
        """
        # Pattern: Simple search implementation (would be more sophisticated in production)
        results = []
        all_articles = self._get_all_articles()

        for article in all_articles:
            relevance_score = 0

            # Pattern: Search in title (weight: 3)
            if query.lower() in article["title"].lower():
                relevance_score += 3

            # Pattern: Search in content summary (weight: 1)
            if query.lower() in article["content"]["summary"].lower():
                relevance_score += 1

            # Pattern: Filter by category
            if categories and article["category"] not in categories:
                continue

            # Pattern: Filter by tags
            if tags and not any(tag in article["tags"] for tag in tags):
                continue

            if relevance_score > 0:
                results.append({
                    "article_id": article["article_id"],
                    "title": article["title"],
                    "category": article["category"],
                    "summary": article["content"]["summary"],
                    "relevance_score": relevance_score
                })

        # Pattern: Sort by relevance
        results.sort(key=lambda x: x["relevance_score"], reverse=True)

        return results
